FSM Raspberry Pi

I started implementing the elevator's Finite State Machine on the Raspberry Pi Supervisory Controller. The FSM tracks whether the elevator is sitting at one of the three floors or moving, and keeps separate priority queues for car, floor, and website requests. Below are the core data structures and the initialization routine.


typedef enum {
    STATE_FLOOR1 = 0,   // floor number - 1, used as an array index
    STATE_FLOOR2 = 1,
    STATE_FLOOR3 = 2,
    STATE_MOVING = 3
} ElevatorSate;

typedef struct {
    ElevatorSate state;

    // Priority queues
    std::queue<int> carQueue;      // Priority 1
    std::queue<int> floorQueue;    // Priority 2
    std::queue<int> websiteQueue;  // Priority 3

    int doorClosed;        // 0 = door open, 1 = door closed
    int previousFloor;     // last floor the elevator was AT before moving
    int targetFloor;       // destination floor while state == STATE_MOVING

    time_t doorTimeStart;  // timestamp when the door was last opened
} ElevatorFSM;

// Initial Condition
void fsmInit(ElevatorFSM *fsm) {
    fsm->state = STATE_FLOOR1;

    fsm->carQueue = std::queue<int>();
    fsm->floorQueue = std::queue<int>();
    fsm->websiteQueue = std::queue<int>();

    fsm->doorClosed = 0;   // door starts open
    fsm->previousFloor = 1;
    fsm->targetFloor = 1;
    fsm->doorTimeStart = time(NULL);
}
